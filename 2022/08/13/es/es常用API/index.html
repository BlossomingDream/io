        int size = 10000;
        while (true) {
            //建立连接
            BulkRequest bulkRequest = new BulkRequest();
            //获取数据
            PageDTO<Item> itemList = itemClient.getItemByPage( page, size );

            if ( list.size() <= 0 ) {
                break;
            }
            //循环得到每条数据

                indexRequest.id(item.getId().toString())

                bulkRequest.add( indexRequest );

            }
            client.bulk( bulkRequest, RequestOptions.DEFAULT );
            page++;
        }
        log.error( "数据插入成功" );
    }

es中常用条件查询

/**
    * list查询条件
    * 布尔查询条件 -- 结果是一组组数据
    *  条件查询
    * @param request
    * @param params
    */
   private SearchRequest conditionByList ( SearchRequest request, RequestParams params ) {
       String key = params.getKey();

       Integer page = params.getPage();
       Integer size = params.getSize();
       String brand = params.getBrand();
       String sortBy = params.getSortBy();
       String category = params.getCategory();
       String minPrice = params.getMinPrice();
       String maxPrice = params.getMaxPrice();
       //创建布尔查询连接
       BoolQueryBuilder boolQuery = new BoolQueryBuilder();

       //搜索框    
       if ( key == null || "".equals( key ) ) {
           //搜索框无条件 - 查询所有
           boolQuery.must( QueryBuilders.matchAllQuery() );
       } else {
           //搜索框有输入 - 根据字段搜索
           boolQuery.must( QueryBuilders.matchQuery( "all", key ) );
       }

       //根据条件分组聚合搜索 -- 实现分组结果
       //1.品牌分组
       if ( brand != null && !"".equals( brand ) ) {
           boolQuery.must(
                   //品牌分组
                   QueryBuilders.termQuery( "brand", brand )

           );
       }
       //2.种类分组
       if ( category != null && !"".equals( category ) ) {
           //精确字段
           boolQuery.must( QueryBuilders.termQuery( "category", category ) );
       }

       //3.分页展示数据
       request.source().query( boolQuery )
               .from( (page - 1) * size )                       //记录起始下标
               .size( size );                    //每页大小
       //4.排序
       if ( "sold".equals( sortBy ) ) {
           //销量降序
           request.source().query( boolQuery )
                   .sort( sortBy, SortOrder.DESC );
       } else if ( "price".equals( sortBy ) ) {
           //价钱升序
           request.source().query( boolQuery )
                   .sort( sortBy, SortOrder.ASC );
       }
       //5.价钱范围查询
       if ( maxPrice != null && minPrice != null ) {
           boolQuery.filter(
                   QueryBuilders.rangeQuery( "price" )
                           //大于等于          小于等于
                           .gte( minPrice ).lte( maxPrice )
           );
       }

       //6.设置高亮

       request.source().highlighter(
               new HighlightBuilder()
                       .field( "name" ).requireFieldMatch( false )
       );
       //7.算分排序    广告
       FunctionScoreQueryBuilder functionScoreQueryBuilder = QueryBuilders.functionScoreQuery( new FunctionScoreQueryBuilder(
               //原始查询
               boolQuery,
               //function score数组
               new FunctionScoreQueryBuilder.FilterFunctionBuilder[]{
                       //其中的每一个function元素
                       new FunctionScoreQueryBuilder.FilterFunctionBuilder(
                               //过滤条件
                               QueryBuilders.termQuery( "isAD", true ),
                               //策略
                               ScoreFunctionBuilders.weightFactorFunction( 10 )
                       )
               }

       ) );


       request.source().query( functionScoreQueryBuilder );
       return request;
   }

解析条件查询后的结果

/**
    * 解析结果方法
    */
   private PageDTO<ItemResult> parseResult ( SearchRequest request ) {
       Long value = null;
       ArrayList<ItemResult> list = null;
       try {
           //解析结果
           SearchResponse response = client.search( request, RequestOptions.DEFAULT );
           //获取hits
           SearchHits searchHits = response.getHits();
           //获取条目
           value = searchHits.getTotalHits().value;
           //获取hits中的每一组数据 -- doc数组
           SearchHit[] hits = searchHits.getHits();
           list = new ArrayList<>();
           //遍历每一条doc数组
           for (SearchHit hit : hits) {
               String item = hit.getSourceAsString();
               //反序列化
               ItemResult itemResult = JSON.parseObject( item, ItemResult.class );
               //获取高亮结果
               Map<String, HighlightField> highlightFields = hit.getHighlightFields();
               //判断是否有高亮结果
               if ( !CollectionUtils.isEmpty( highlightFields ) ) {
                   HighlightField highName = highlightFields.get( "name" );
                   //健壮性判断一下
                   if ( highName != null ) {
                       //取出高亮的结果       //高亮结果是数组类型
                       String resultHigh = highName.getFragments()[0].toString();
                       //覆盖name
                       itemResult.setName( resultHigh );
                   }
               }
               list.add( itemResult );
           }
           return PageDTO.setResult( value, list );
       } catch (IOException e) {
           e.printStackTrace();
       }
       return PageDTO.setResult( value, list );
   }

聚合查询

/**
   * 分组布尔查询
   */
  @PostMapping("filters")
  public Map<String, List<String>> getFilters ( @RequestBody RequestParams params ) {
      Integer size = params.getSize();

      //建立请求
      SearchRequest request = new SearchRequest( EsConstant.ES_INDEX );
      //设置聚合条件
      conditionByList( request, params );
      //品牌聚合           //聚合姓名            //聚合分组之后的大小 
      aggByName( request, "聚合姓名", "聚合的字段", size );
      //种类聚合
      aggByName( request, "聚合姓名", "聚合的字段", size );

      HashMap<String, List<String>> map = new HashMap<>();
      try {
          //发送查询请求
          SearchResponse response = client.search( request, RequestOptions.DEFAULT );
          //结果处理
          //品牌结果处理
          List<String> brandList = parseAggResult( response, "聚合姓名" );
          map.put( "key", brandList );
          //种类结果处理
          List<String> categoryList = parseAggResult( response, "聚合姓名" );
          map.put( "key", categoryList );
          return map;
      } catch (IOException e) {
          e.printStackTrace();
      }
      return map;
  }

解析聚合结果

private List<String> parseAggResult ( SearchResponse response, String "聚合姓名" ) {


       //解析聚合结果
       Aggregations aggregations = response.getAggregations();
       //取出agg根据aggName
       Terms brandTerms = aggregations.get( "聚合姓名" );
       //获得桶集合
       List<? extends Terms.Bucket> bucketsBrand = brandTerms.getBuckets();
       //遍历桶
       ArrayList<String> listBrand = new ArrayList<>();
       for (Terms.Bucket bucket : bucketsBrand) {
           //获取key,品牌信息
           String brand = bucket.getKeyAsString();
           //添加到集合
           listBrand.add( brand );
       }
       return listBrand;
   }

搜索框自动补全

自动补全需要要实现根据字母做补全，就必须对文档按照拼音分词。在GitHub上恰好有elasticsearch的拼音分词插件。

   /**
     * 自动补全
     *
     * @param key
     * @return
     */
    @GetMapping("suggestion")
    public List<String> getSuggestion ( @RequestParam("key") String key ) {
        //1.与es建立连接
        //2.设置条件
        //3.解析相应结果
        //4.返回结果

        SearchRequest request = new SearchRequest( "索引库" );
        //设置自动补全查询条件  key==搜索的字段
        condition( request, key );
        ArrayList<String> list = new ArrayList<>();
        //发送请求
        try {
            SearchResponse response = client.search( request, RequestOptions.DEFAULT );
            //解析响应结果
            Suggest suggest = response.getSuggest();
            //根据自动查询补全名称,获取补全结果
            CompletionSuggestion suggestions = suggest.getSuggestion( 
                "设置的自动补全的名字" );
            //获取options
            List<CompletionSuggestion.Entry.Option> options = suggestions.getOptions();
            //遍历结果
            for (CompletionSuggestion.Entry.Option option : options) {
                //取出文档数据
                String text = option.getText().toString();
                //添加到数组中
                list.add( text );
            }
            return list;
        } catch (IOException e) {
            e.printStackTrace();
        }
        return list;
    }

    /**
     * 自动补全条件
     *
     * @param request
     * @param key
     */
    private void condition ( SearchRequest request, String key ) {

        //当搜索框有条件才会设置该条件
        //可以收用StringUtils工具是否为空
        if ( !"".equals( key ) && key != null ) {
            request.source().suggest(
                    new SuggestBuilder().addSuggestion(
                        //设置自动补全查询名字
                            "自动补全的名字",
                            SuggestBuilders.completionSuggestion(   
                                     "suggestion" )
                                    .prefix( key )
                                    .skipDuplicates( true )
                                    .size( 20 )
                    )
            );
        }
    }
