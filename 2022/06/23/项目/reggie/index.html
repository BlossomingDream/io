@Data
@ApiModel("返回结果类")
public class R<T> implements Serializable {
    @ApiModelProperty("编码")
    private Integer code;
    @ApiModelProperty("响应信息")//编码：1成功，0和其它数字为失败
    private String msg;
    @ApiModelProperty("响应数据")//错误信息
    private T data;
    @ApiModelProperty("动态数据")//数据
    private Map map = new HashMap();    //动态数据

    /**
     * 快捷创建R对象，封装成功的提示信息和数据
     * @param msg
     * @param object
     * @param <T>
     * @return
     */
    public static <T> R<T> success(String msg, T object) {
        R<T> r = new R<T>();
        r.msg = msg;
        r.data = object;
        r.code = 1;
        return r;
    }

    /**
     * 快捷创建Result对象，封装成功的提示信息
     * @param msg
     * @param <T>
     * @return
     */
    public static <T> R<T> success(String msg) {
        R<T> r = new R<T>();
        r.msg = msg;
        r.code = 1;
        return r;
    }

    /**
     * 快捷创建Result对象，封装失败的提示信息
     * @param msg
     * @param <T>
     * @return
     */
    public static <T> R<T> error(String msg) {
        R r = new R();
        r.msg = msg;
        r.code = 0;
        return r;
    }

    public R<T> add(String key, Object value) {
        this.map.put(key, value);
        return this;
    }
}

 * 作用:为了解决返回值封装数据代码冗余

                                     /*指定包下所有方法都会被该类增强*/
@RestControllerAdvice(basePackages = "com.itheima.reggie.web.controller")
public class ControllerReturnValueAdvice implements ResponseBodyAdvice<Object> {
    /**
     * 支持。哪些handler需要被统一封装。
     * 返回true，要继续调用beforeBodyWrite方法，会封装；
     * 返回false，不再调用beforeBodyWrite方法，不会封装；
     * <p>
     * 一般情况下，会在该方法内部判断哪些handler需要封装，需要封装的返回true；否则，返回false。
     * 本案中，为了简化处理，直接返回true；具体区分逻辑在beforeBodyWrite中实现
     */
    @Override
    public boolean supports ( MethodParameter methodParameter, Class<? extends HttpMessageConverter<?>> aClass ) {

        return true;
    }

    @Override
    public Object beforeBodyWrite ( Object returnValue, MethodParameter methodParameter, MediaType mediaType, Class<? extends HttpMessageConverter<?>> aClass, ServerHttpRequest serverHttpRequest, ServerHttpResponse serverHttpResponse ) {
        //1.对返回值的类型判断 若是R
        if ( returnValue instanceof R ) {
            //1.1直接return
            return returnValue;
        }

        //2.若返回值类型是String类型,需要手动序列化成对象响应前端
        if ( returnValue instanceof String ) {
            R<Object> r = R.success( (String) returnValue );
            //2.1创建序列化对象将字符串对象序列化
            //2.2像防御fastJSon序列化核心对象
            ObjectMapper objectMapper = new ObjectMapper();
            String value = "";
            try {
                //2.3相当于JSON.toString
                value = objectMapper.writeValueAsString( r );
            } catch (JsonProcessingException e) {
                e.printStackTrace();
            }
            return value;
        }

        //3其他情况直接封装后返回
        if ( returnValue == null || Boolean.valueOf( false ).equals( returnValue ) ) {
            return R.error( "咋回事?再试一次?" );
        }
        //4.程序执行到这里表示返回的既不是 R,也不是 String ,也不是 false
        //4.1将数据封装后返回
        return R.success( "ok", returnValue );
    }
}

//过滤器
@WebFilter("/*")    //对所有资源进行拦截     //过滤器注解
@Slf4j
public class LoginFilter implements Filter {

    //创建路径匹配器
    private final AntPathMatcher apm = new AntPathMatcher();
    //定义可以访问的url
    @Value("${accessUrl}")
    private String accessUrl;


    @Override
    public void doFilter ( ServletRequest request, ServletResponse response, FilterChain chain ) throws IOException, ServletException {
        HttpServletRequest req = (HttpServletRequest) request;
        //1.获取uri
        String requestURI = req.getRequestURI();
        log.info( "用户访问url: " + requestURI );
        //对用户访问的url进行判断
        /*if ( ||"index".equals( requestURI ) ){
            //访问的是登录相关 放心
            chain.doFilter( req, response );
            return;
        }*/

        //2.对配置文件获取的可访问路径进行处理
        String[] split = accessUrl.split( "," );
        if ( checkUrl( requestURI, split ) ) {
            //2.1表明访问的是可访问资源
            //2.2放行
            chain.doFilter( req, response );
            return;
        }

        //3用户访问的不是登录相关的
        Object employeeId = req.getSession().getAttribute( "usernameId" );
        
        //3.1对用户是否登录进行判断
        if ( employeeId != null ) {
            // TODO: 2022/7/11 将id存入线程变量中 :要在放行前存入.每一次请求只会有同一线程来进入
            // TODO: 2022/7/11 将id存入此线程变量实现id解耦
            ThreadLocalUtils.setCurrentId( (Long) employeeId );
            log.info( "线程名称为: {}", Thread.currentThread() );
            //3.2表示用户已经退出,跳转登录
            chain.doFilter( req, response );
            return;
        }
        //4用户登录放行
        // 4.1 判断登录状态，如果已登录，则直接放行
        Object userId = req.getSession().getAttribute("userId");

        if (userId != null) {
            // 如果用户登录了，不光要直接放行，
            // // 还要在放行前把用户id设置进线程绑定的容器中
            ThreadLocalUtils.setCurrentId((Long) userId);

            // 登录后放行
            chain.doFilter( req, response );

            // 后面还要判断登录状态，已经放行的内容不需要再次判断登录状态，直接返回
            return;
        }


        //4.用户既没有登录又访问的不是放行的资源
        //4.1向前台响应notLogin
        R<Object> error = R.error( "NOTLOGIN" );

        //4.2转为json响应前台
        String toJSONString = JSON.toJSONString( error );
        response.getWriter().write( toJSONString );

    }

    private boolean checkUrl ( String url, String[] urls ) {
        //对访问路径进行判断
        //循环可访问url
        for (String s : urls) {

            //1.参数1:被检查的url 参数2:url集合   return 是否包含
            boolean match = apm.match( s, url );

            //2.如果包含返回true,表明访问的url是可访问的
            if ( match ) {
                return true;
            }
        }

        //3.访问的是没有放行的资源
        return false;
    }
}

package com.itheima.reggie.utils;

import java.io.File;
import java.util.UUID;

/**
 * 文件命名工具类
 *
 * @Author WeiFanQuan
 * @Date 2022/7/12 21:16
 * @Emil 8238504@qq.com
 * @Description: 文件命名工具类
 * 根据生成的文件命名的前三位创建不同的三层文件夹  a-z + 10 位数 创建三层文件夹
 */
public class FileUtils {
    /**
     * 根据生成的文件命名的前三位创建不同的三层文件夹  a-z + 10 位数 创建三层文件夹
     * @return
     */
    public static String[] getFileNameAndPath () {
        //创建文件命名
        String filename = UUID.randomUUID().toString();
        filename.replace( "-", "" );

        //获取前三个字符作为目录名称
        char c1 = filename.charAt( 0 );
        char c2 = filename.charAt( 1 );
        char c3 = filename.charAt( 2 );

        //拼接路径
        StringBuilder filepath = new StringBuilder().append( c1 ).
                append( "/" ).append( c2 ).append( "/" ).append( c3 ).append( "/" );
        //返回三层路径名与文件明
        return new String[]{filepath.toString(), filename};
    }

    /**
     * 生成包含三级目录的随机文件名
     *
     * @return 文件名，包含三级目录
     */
    public static String getFileNameWithPath() {
        // 生成一个唯一的文件名
        String fineName = UUID.randomUUID().toString();
        fineName = fineName.replaceAll("-", "");

        // 获取前三个字符作为目录名
        char c1 = fineName.charAt(0);
        char c2 = fineName.charAt(1);
        char c3 = fineName.charAt(2);

        StringBuilder filePath = new StringBuilder().append(c1).append("/").append(c2).append("/").append(c3).append("/");

        return filePath.append(fineName).toString();
    }



    /**
     * 文件路径是否存在,不存在则创建
     * @param fileNameWithPath: 子路径 三层路径
     * @param uploadRealPath: 基础父路径 F:img/
     */
    public static void makeDirs ( String fileNameWithPath, String uploadRealPath ) {
        //判断目录是否存在
        String filepath = fileNameWithPath.substring( 0, fileNameWithPath.lastIndexOf( "/" ) );
        //若不存则在创建   //param:父路径     param:子路径
        if ( !new File( uploadRealPath, filepath ).exists() ) {
            new File( uploadRealPath, filepath ).mkdirs();
        }
    }

}

/**
 * 文件上传控制类

@RestController
@RequestMapping("/common")
@Slf4j
public class CommonFileUpload {

    @Value("${basepath}")
    private String basepath;

    /**
     * 图片上传
     *
     * @param file :
     * @return
     * @throws IOException
     */
    @PostMapping("/upload")
    public R uplodaFile ( MultipartFile file ) throws IOException {
        //获取文件源名
        String originalFilename = file.getOriginalFilename();

        //获取文件后缀
        String substring = originalFilename.substring( originalFilename.lastIndexOf( "." ) );

        //获取新命名和路径
        String fileNameWithPath = FileUtils.getFileNameWithPath();

        //调用工具判断该路径是否存在,不存在则创建
        FileUtils.makeDirs( fileNameWithPath, basepath );

        //上床的文件存到指定位置      param:父路径     param:子路径
        file.transferTo( new File( basepath, fileNameWithPath + "." + substring ) );

        return R.success( "ok", fileNameWithPath + "." + substring );
    }

    /**
     * 文件回显下载
     *
     * @param name
     * @param resp
     * @throws IOException
     */
    @GetMapping("/download")
    public void downloadFile ( String name, HttpServletResponse resp ) throws IOException {
        //获取网络输出流
        ServletOutputStream os = resp.getOutputStream();

        //转为字节缓冲输出流
        BufferedOutputStream bos = new BufferedOutputStream( os );

        //创建字节换缓冲输入流
        BufferedInputStream bis = new BufferedInputStream( new FileInputStream( basepath + name ) );

        //定义传输数组
        byte[] bys = new byte[1024];

        //定义读取字节
        int len;

        //读取文件只要是-1表示文件已经读取完毕
        while ((len = bis.read( bys )) != -1) {
            //  bys:内存中字节数组传递  len:读取同步
            bos.write( bys, 0, len );
            bos.flush();
        }
        bis.close();
        os.close();
    }

}

    /**
     * 扩展mvc框架的消息转换器
     *
     * @param converters 转换器列表集合
     */
    @Override
    protected void extendMessageConverters( List<HttpMessageConverter<?>> converters) {
        // 创建消息转换器对象
        MappingJackson2HttpMessageConverter messageConverter = new MappingJackson2HttpMessageConverter();
        // 设置自定义的对象映射器
        messageConverter.setObjectMapper(new OptimizationObjectMapper());
        // 通过设置索引，让自己的转换器放在最前面，否则默认的jackson转换器会在前面，用不上自己配置的转换器
        converters.add(0, messageConverter);
    }


    /**
     * 定义成员内部类，实现该功能
     * <p>
     * 对象映射器：指定序列化和反序列化时数据映射格式。
     * 本对象映射器继承自Jackson中提供的转换核心映射器ObjectMapper
     * 支持
     * 指定格式的日期时间字符串与LocalDateTime的序列化和反序列化
     * BigInteger、Long --> String的序列化
     */

     class OptimizationObjectMapper extends ObjectMapper {
        public static final String DEFAULT_DATE_FORMAT = "yyyy-MM-dd";
        public static final String DEFAULT_DATE_TIME_FORMAT = "yyyy-MM-dd HH:mm:ss";
        public static final String DEFAULT_TIME_FORMAT = "HH:mm:ss";

        public OptimizationObjectMapper() {
            super();
            // 反序列化特性：反序列化时属性不存在的兼容处理，未知属性时不报异常
            this.configure(FAIL_ON_UNKNOWN_PROPERTIES, false);
            this.getDeserializationConfig().withoutFeatures(FAIL_ON_UNKNOWN_PROPERTIES);

            // 新建功能模块，并添加序列化器和反序列化器
            SimpleModule simpleModule = new SimpleModule()
                    // 添加指定规则的反序列化器
                    .addDeserializer( LocalDateTime.class, new LocalDateTimeDeserializer( DateTimeFormatter.ofPattern(DEFAULT_DATE_TIME_FORMAT)))
                    .addDeserializer( LocalDate.class, new LocalDateDeserializer(DateTimeFormatter.ofPattern(DEFAULT_DATE_FORMAT)))
                    .addDeserializer( LocalTime.class, new LocalTimeDeserializer(DateTimeFormatter.ofPattern(DEFAULT_TIME_FORMAT)))

                    // 添加指定规则的序列化器
                    .addSerializer( BigInteger.class, ToStringSerializer.instance)
                    // Long --> String，可以解决本案中的问题
                    .addSerializer(Long.class, ToStringSerializer.instance)

                    .addSerializer(LocalDateTime.class, new LocalDateTimeSerializer(DateTimeFormatter.ofPattern(DEFAULT_DATE_TIME_FORMAT)))
                    .addSerializer(LocalDate.class, new LocalDateSerializer(DateTimeFormatter.ofPattern(DEFAULT_DATE_FORMAT)))
                    .addSerializer(LocalTime.class, new LocalTimeSerializer(DateTimeFormatter.ofPattern(DEFAULT_TIME_FORMAT)));
            //注册功能模块
            this.registerModule(simpleModule);
        }
    }


}

**
 * 随机生成验证码工具类
 */
public class ValidateCodeUtils {
    /**
     * 随机生成验证码
     * @param length 长度为4位或者6位
     * @return
     */
    public static Integer generateValidateCode(int length){
        Integer code =null;
        if(length == 4){
            code = new Random().nextInt(9999);//生成随机数，最大为9999
            if(code < 1000){
                code = code + 1000;//保证随机数为4位数字
            }
        }else if(length == 6){
            code = new Random().nextInt(999999);//生成随机数，最大为999999
            if(code < 100000){
                code = code + 100000;//保证随机数为6位数字
            }
        }else{
            throw new RuntimeException("只能生成4位或6位数字验证码");
        }
        return code;
    }

    /**
     * 随机生成指定长度字符串验证码
     * @param length 长度
     * @return
     */
    public static String generateValidateCode4String(int length){
        Random rdm = new Random();
        String hash1 = Integer.toHexString(rdm.nextInt());
        String capstr = hash1.substring(0, length);
        return capstr;
    }
}
